# 2. 빌드 파일 복사 및 커밋
cp -r build/* .
git add .
git commit -m "Deploy: Production build v2.1.2 with hierarchical evaluation system"

git push origin gh-pages-deploy

# 4. 메인 브랜치로 복귀
git checkout main
